<?php

class RootObject
{
	/** @var string[] */
	public $stringArray;
	/** @var (string|int|null)[] */
	public $mixedArray;
	public $nullArray;
	public $unknownArray;
	public $boolean;
	public $float;
	public $int;
	public $null;
	public $nestedClass;
	public $string;

	public function toArray(): array
	{
		return [
			'string_array' => $this->stringArray,
			'mixed_array' => $this->mixedArray,
			'null_array' => $this->nullArray,
			'unknown_array' => $this->unknownArray,
			'boolean' => $this->boolean,
			'float' => $this->float,
			'int' => $this->int,
			'null' => $this->null,
			'nested_class' => $this->nestedClass->toArray(),
			'string' => $this->string
		];
	}
}

<?php

class NestedClass
{
	public $var;

	public function toArray(): array
	{
		return [
			'var' => $this->var
		];
	}
}
