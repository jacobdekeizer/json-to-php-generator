<?php

class RootObject
{
	/** @var string[] */
	public array $stringArray;
	/** @var (string|int|null)[] */
	public array $mixedArray;
	public array $nullArray;
	public array $unknownArray;
	public bool $boolean;
	public float $float;
	public int $int;
	public null $null;
	public NestedClass $nestedClass;
	public string $string;

	public function toArray(): array
	{
		return [
			'string_array' => $this->string_array,
			'mixed_array' => $this->mixed_array,
			'null_array' => $this->null_array,
			'unknown_array' => $this->unknown_array,
			'boolean' => $this->boolean,
			'float' => $this->float,
			'int' => $this->int,
			'null' => $this->null,
			'nested_class' => $this->nested_class->toArray(),
			'string' => $this->string
		];
	}
}

<?php

class NestedClass
{
	public int $var;

	public function toArray(): array
	{
		return [
			'var' => $this->var
		];
	}
}
