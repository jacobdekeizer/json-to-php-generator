<?php

class RootObject
{
	/** @var SimpleNesting[] */
	public $simpleNesting;

	public function toArray(): array
	{
		return [
			'simple_nesting' => array_map(static function($item) {
				    return $item->toArray();
			}, $this->simpleNesting)
		];
	}
}

<?php

class SimpleNesting
{
	public $alwaysPresent;
	public $string;
	public $object;
	public $number;

	public function toArray(): array
	{
		return [
			'always_present' => $this->alwaysPresent,
			'string' => $this->string,
			'object' => $this->object !== null ? $this->object->toArray() : null,
			'number' => $this->number
		];
	}
}

<?php

class Object
{
	public $test;
	public $anotherTest;

	public function toArray(): array
	{
		return [
			'test' => $this->test,
			'another_test' => $this->anotherTest
		];
	}
}
